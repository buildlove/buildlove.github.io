mysql

初始化mysql密码

PATH=”$PATH”:/usr/local/mysql/bin
cd usr/local/mysql/bin
sudo su

跳过mysql验证

./mysqld_safe –skip-grant-tables

打开mysql，修改密码

./mysql
FLUSH PRIVILEGES;
SET PASSWORD FOR ‘root’@’localhost’ = PASSWORD(‘你的新密码’);

使用账号密码登陆

mysql -u root -p(“你的密码”)
